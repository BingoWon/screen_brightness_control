  16    '''
  17    A way of getting display information and adjusting the brightness
  18    that does not rely on any 3rd party software.
  19
  20    This class works with displays that show up in the `/sys/class/backlight`
  21    directory (so usually laptop displays).
  22

  24    `/sys/class/backlight/*/brightness` or you will need to run the program
  25    as root.

  28    def get_display_info(cls, display: Optional[Union[int, str]] = None) -> List[dict]:
  29        '''
  30        Returns information about detected displays by reading files from the
  31        `/sys/class/backlight` directory
  32
  33        Args:
  34            display (str or int): [*Optional*] The monitor to return info about.
  35                Pass in the serial number, name, model, interface, edid or index.
  36                This is passed to `filter_monitors`
  37
  38        Returns:
  39            list: list of dicts
  40
  41        Example:
  42            ```python
  43            import screen_brightness_control as sbc
  44
  45            # get info about all monitors
  46            info = sbc.linux.SysFiles.get_display_info()
  47            # EG output: [{'name': 'edp-backlight', 'path': '/sys/class/backlight/edp-backlight', edid': '00ffff...'}]
  48
  49            # get info about the primary monitor
  50            primary_info = sbc.linux.SysFiles.get_display_info(0)[0]
  51
  52            # get info about a monitor called 'edp-backlight'
  53            edp_info = sbc.linux.SysFiles.get_display_info('edp-backlight')[0]
  54            ```
  55        '''
  56        subsystems = set()
  57        for folder in os.listdir('/sys/class/backlight'):
  58            if os.path.isdir(f'/sys/class/backlight/{folder}/subsystem'):
  59                subsystems.add(tuple(os.listdir(f'/sys/class/backlight/{folder}/subsystem')))
  60
  61        all_displays = {}
  62        index = 0
  63
  64        for subsystem in subsystems:
  65
  66            device = {
  67                'name': subsystem[0],
  68                'path': f'/sys/class/backlight/{subsystem[0]}',
  69                'method': cls,
  70                'index': index,
  71                'model': None,
  72                'serial': None,
  73                'manufacturer': None,
  74                'manufacturer_id': None,
  75                'edid': None,
  76                'scale': None
  77            }
  78

  80                # subsystems like intel_backlight usually have an acpi_video0
  81                # counterpart, which we don't want so lets find the 'best' candidate
  82                try:
  83                    with open(os.path.join(f'/sys/class/backlight/{folder}/max_brightness')) as f:
  84                        scale = int(f.read().rstrip(' \n')) / 100

  87                    if device['scale'] is None or scale > device['scale']:
  88                        device['name'] = folder

 235    class I2CDevice():
 236        '''
 237        Class to read and write data to an I2C bus,
 238        based on the `I2CDev` class from [ddcci.py](https://github.com/siemer/ddcci)
 239        '''
 240        def __init__(self, fname: str, slave_addr: int):
 241            '''
 242            Args:
 243                fname (str): the I2C path, eg: `/dev/i2c-2`
 244                slave_addr (int): not entirely sure what this is meant to be
 245            '''
 246            self.device = os.open(fname, os.O_RDWR)
 247            # I2C_SLAVE address setup
 248            fcntl.ioctl(self.device, I2C.I2C_SLAVE, slave_addr)
 249
 250        def read(self, length: int) -> bytes:
 251            '''
 252            Read a certain number of bytes from the I2C bus
 253
 254            Args:
 255                length (int): the number of bytes to read
 256
 257            Returns:
 258                bytes
 259            '''
 260            return os.read(self.device, length)
 261
 262        def write(self, data: bytes) -> int:
 263            '''
 264            Writes data to the I2C bus
 265
 266            Args:
 267                data (bytes): the data to write
 268
 269            Returns:
 270                int: the number of bytes written
 271            '''
 272            return os.write(self.device, data)

 274    class DDCInterface(I2CDevice):
 275        '''
 276        Class to send DDC (Display Data Channel) commands to an I2C device,
 277        based on the `Ddcci` and `Mccs` classes from [ddcci.py](https://github.com/siemer/ddcci)
 278        '''
 279
 280        PROTOCOL_FLAG = 0x80
 281
 282        def __init__(self, i2c_path: str):
 283            '''
 284            Args:
 285                i2c_path (str): the path to the I2C device, eg: `/dev/i2c-2`
 286            '''
 287            super().__init__(i2c_path, I2C.DDCCI_ADDR)
 288
 289        def write(self, *args) -> int:
 290            '''
 291            Write some data to the I2C device.
 292
 293            It is recommended to use `setvcp` to set VCP values on the DDC device
 294            instead of using this function directly.
 295
 296            Args:
 297                *args: variable length list of arguments. This will be put
 298                    into a `bytearray` and wrapped up in various flags and
 299                    checksums before being written to the I2C device
 300
 301            Returns:
 302                int: the number of bytes that were written
 303            '''
 304            time.sleep(I2C.WAIT_TIME)
 305
 306            ba = bytearray(args)
 307            ba.insert(0, len(ba) | self.PROTOCOL_FLAG)  # add length info
 308            ba.insert(0, I2C.HOST_ADDR_W)  # insert source address
 309            ba.append(functools.reduce(operator.xor, ba, I2C.DESTINATION_ADDR_W))  # checksum
 310
 311            return super().write(ba)
 312
 313        def setvcp(self, vcp_code: int, value: int) -> int:
 314            '''
 315            Set a VCP value on the device
 316
 317            Args:
 318                vcp_code (int): the VCP command to send, eg: `0x10` is brightness
 319                value (int): what to set the value to
 320
 321            Returns:
 322                int: the number of bytes written to the device
 323            '''
 324            return self.write(I2C.SET_VCP_CMD, vcp_code, *value.to_bytes(2, 'big'))
 325
 326        def read(self, amount: int) -> bytes:
 327            '''
 328            Reads data from the DDC device.
 329
 330            It is recommended to use `getvcp` to retrieve VCP values from the
 331            DDC device instead of using this function directly.
 332
 333            Args:
 334                amount (int): the number of bytes to read
 335
 336            Returns:
 337                bytes
 338
 339            Raises:
 340                ValueError: if the read data is deemed invalid
 341            '''
 342            time.sleep(I2C.WAIT_TIME)
 343
 344            ba = super().read(amount + 3)
 345
 346            # check the bytes read
 347            checks = {
 348                'source address': ba[0] == I2C.DESTINATION_ADDR_W,
 349                'checksum': functools.reduce(operator.xor, ba) == I2C.HOST_ADDR_R,
 350                'length': len(ba) >= (ba[1] & ~self.PROTOCOL_FLAG) + 3
 351            }
 352            if False in checks.values():
 353                log.debug('i2c read check failed: ' + repr(checks))
 354                raise ValueError('i2c read check failed: ' + repr(checks))
 355
 356            return ba[2:-1]
 357
 358        def getvcp(self, vcp_code: int) -> Tuple[int, int]:
 359            '''
 360            Retrieves a VCP value from the DDC device.
 361
 362            Args:
 363                vcp_code (int): the VCP value to read, eg: `0x10` is brightness
 364
 365            Returns:
 366                tuple[int, int]: the current and maximum value respectively
 367
 368            Raises:
 369                ValueError: if the read data is deemed invalid
 370            '''
 371            self.write(I2C.GET_VCP_CMD, vcp_code)
 372            ba = self.read(8)
 373
 374            checks = {
 375                'is feature reply': ba[0] == I2C.GET_VCP_REPLY,
 376                'supported VCP opcode': ba[1] == 0,
 377                'answer matches request': ba[2] == vcp_code
 378            }
 379            if False in checks.values():
 380                log.debug('i2c read check failed: ' + repr(checks))
 381                raise ValueError('i2c read check failed: ' + repr(checks))
 382
 383            # current and max values
 384            return int.from_bytes(ba[6:8], 'big'), int.from_bytes(ba[4:6], 'big')

 386    @classmethod
 387    def get_display_info(cls, display: Optional[Union[int, str]] = None) -> List[dict]:
 388        '''
 389        Returns information about detected displays by querying the various I2C buses
 390
 391        Args:
 392            display (str or int): [*Optional*] The monitor to return info about.
 393                Pass in the serial number, name, model, interface, edid or index.
 394                This is passed to `filter_monitors`
 395
 396        Returns:
 397            list: list of dicts
 398
 399        Example:
 400            ```python
 401            import screen_brightness_control as sbc
 402
 403            # get info about all monitors
 404            info = sbc.linux.I2C.get_display_info()
 405            # EG output: [{'name': 'Benq GL2450H', 'model': 'GL2450H', 'manufacturer': 'BenQ', 'edid': '00ffff...'}]
 406
 407            # get info about the primary monitor
 408            primary_info = sbc.linux.I2C.get_display_info(0)[0]
 409
 410            # get info about a monitor called 'Benq GL2450H'
 411            benq_info = sbc.linux.I2C.get_display_info('Benq GL2450H')[0]
 412            ```
 413        '''
 414        all_displays = __cache__.get('i2c_display_info')
 415        if all_displays is None:
 416            all_displays = []
 417            index = 0
 418
 419            for i2c_path in glob.glob('/dev/i2c-*'):
 420                if not os.path.exists(i2c_path):
 421                    continue
 422
 423                try:
 424                    # open the I2C device using the host read address
 425                    device = cls.I2CDevice(i2c_path, cls.HOST_ADDR_R)
 426                    # read some 512 bytes from the device
 427                    data = device.read(512)
 428                except IOError as e:
 429                    log.debug(f'I2C: IOError reading from device {i2c_path}: {e}')
 430                    continue
 431
 432                # search for the EDID header within our 512 read bytes
 433                start = data.find(bytes.fromhex('00 FF FF FF FF FF FF 00'))
 434                if start < 0:
 435                    continue
 436
 437                # grab 128 bytes of the edid
 438                edid = data[start: start + 128]
 439                # parse the EDID
 440                manufacturer_id, manufacturer, model, name, serial = EDID.parse(edid)
 441                # convert edid to hex string
 442                edid = ''.join(f'{i:02x}' for i in edid)
 443
 444                all_displays.append(
 445                    {
 446                        'name': name,
 447                        'model': model,
 448                        'manufacturer': manufacturer,
 449                        'manufacturer_id': manufacturer_id,
 450                        'serial': serial,
 451                        'method': cls,
 452                        'index': index,
 453                        'edid': edid,
 454                        'i2c_bus': i2c_path
 455                    }
 456                )
 457                index += 1
 458
 459            if all_displays:
 460                __cache__.store('i2c_display_info', all_displays, expires=2)
 461
 462        if display is not None:
 463            return filter_monitors(display=display, haystack=all_displays, include=['i2c_bus'])
 464        return all_displays

 467    def get_brightness(cls, display: Optional[int] = None) -> List[int]:
 468        '''
 469        Gets the brightness for a display by querying the I2C bus
 470
 471        Args:
 472            display (int): The specific display you wish to query.
 473
 474        Returns:
 475            list: list of ints (0 to 100)
 476
 477        Example:
 478            ```python
 479            import screen_brightness_control as sbc
 480
 481            # get the current display brightness
 482            current_brightness = sbc.linux.I2C.get_brightness()
 483
 484            # get the brightness of the primary display
 485            primary_brightness = sbc.linux.I2C.get_brightness(display = 0)[0]
 486
 487            # get the brightness of the secondary display
 488            secondary_brightness = sbc.linux.I2C.get_brightness(display = 1)[0]
 489            ```
 490        '''
 491        all_displays = cls.get_display_info()
 492        if display is not None:
 493            all_displays = [all_displays[display]]
 494
 495        results = []
 496        for device in all_displays:
 497            interface = cls.DDCInterface(device['i2c_bus'])
 498            value, max_value = interface.getvcp(0x10)
 499
 500            # make sure display's max brighness is cached
 501            cache_ident = '%s-%s-%s' % (device['name'], device['model'], device['serial'])
 502            if cache_ident not in cls._max_brightness_cache:
 503                cls._max_brightness_cache[cache_ident] = max_value
 504
 505            if max_value != 100:
 506                # if max value is not 100 then we have to adjust the scale to be
 507                # a percentage
 508                value = int((value / max_value) * 100)
 509
 510            results.append(value)
 511
 512        return results

276        '''
277        Class to send DDC (Display Data Channel) commands to an I2C device,

281        PROTOCOL_FLAG = 0x80

283        def __init__(self, i2c_path: str):
284            '''
285            Args:

288            super().__init__(i2c_path, I2C.DDCCI_ADDR)

290        def write(self, *args) -> int:
291            '''
292            Write some data to the I2C device.
293
294            It is recommended to use `setvcp` to set VCP values on the DDC device
295            instead of using this function directly.
296
297            Args:
298                *args: variable length list of arguments. This will be put
299                    into a `bytearray` and wrapped up in various flags and
300                    checksums before being written to the I2C device
301
302            Returns:
303                int: the number of bytes that were written
304            '''
305            time.sleep(I2C.WAIT_TIME)
306
307            ba = bytearray(args)
308            ba.insert(0, len(ba) | self.PROTOCOL_FLAG)  # add length info
309            ba.insert(0, I2C.HOST_ADDR_W)  # insert source address
310            ba.append(functools.reduce(operator.xor, ba, I2C.DESTINATION_ADDR_W))  # checksum
311
312            return super().write(ba)

314        def setvcp(self, vcp_code: int, value: int) -> int:
315            '''
316            Set a VCP value on the device
317

320                value (int): what to set the value to
321
322            Returns:
323                int: the number of bytes written to the device
324            '''
325            return self.write(I2C.SET_VCP_CMD, vcp_code, *value.to_bytes(2, 'big'))

327        def read(self, amount: int) -> bytes:
328            '''
329            Reads data from the DDC device.
330
331            It is recommended to use `getvcp` to retrieve VCP values from the
332            DDC device instead of using this function directly.
333
334            Args:
335                amount (int): the number of bytes to read
336
337            Returns:
338                bytes
339
340            Raises:
341                ValueError: if the read data is deemed invalid
342            '''
343            time.sleep(I2C.WAIT_TIME)
344
345            ba = super().read(amount + 3)
346
347            # check the bytes read
348            checks = {
349                'source address': ba[0] == I2C.DESTINATION_ADDR_W,
350                'checksum': functools.reduce(operator.xor, ba) == I2C.HOST_ADDR_R,
351                'length': len(ba) >= (ba[1] & ~self.PROTOCOL_FLAG) + 3
352            }
353            if False in checks.values():
354                log.debug('i2c read check failed: ' + repr(checks))
355                raise ValueError('i2c read check failed: ' + repr(checks))
356
357            return ba[2:-1]

359        def getvcp(self, vcp_code: int) -> Tuple[int, int]:
360            '''
361            Retrieves a VCP value from the DDC device.
362
363            Args:
364                vcp_code (int): the VCP value to read, eg: `0x10` is brightness
365
366            Returns:
367                tuple[int, int]: the current and maximum value respectively
368
369            Raises:
370                ValueError: if the read data is deemed invalid
371            '''
372            self.write(I2C.GET_VCP_CMD, vcp_code)
373            ba = self.read(8)
374
375            checks = {
376                'is feature reply': ba[0] == I2C.GET_VCP_REPLY,
377                'supported VCP opcode': ba[1] == 0,
378                'answer matches request': ba[2] == vcp_code
379            }
380            if False in checks.values():
381                log.debug('i2c read check failed: ' + repr(checks))
382                raise ValueError('i2c read check failed: ' + repr(checks))
383
384            # current and max values
385            return int.from_bytes(ba[6:8], 'big'), int.from_bytes(ba[4:6], 'big')
